### 思路：遍历，使用对象缓存之前值的映射，然后校验是否已存在，则删除
```javascript
var removeDuplicates = function (nums) {
  // 判空 好习惯
  if (!nums || !nums.length) return 0;

  var mapping = {};
  for (var i = 0; i < nums.length; i++) {
    // 校验值是否存在
    if (mapping[nums[i]]) {
      // 删除原数组数据
      nums.splice(i, 1);
      // 索引减 1
      i--;
    }
    // 缓存值的映射
    mapping[nums[i]] = true;
  }

  return nums.length;
};
```
